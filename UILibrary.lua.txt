local UserInputService = game:GetService("UserInputService")
local TweenService     = game:GetService("TweenService")
local Players          = game:GetService("Players")

local function GetGuiParent()
    if type(gethui) == "function" then
        local ok, result = pcall(gethui)
        if ok and result then return result end
    end

    local ok, coreGui = pcall(function() return game:GetService("CoreGui") end)
    if ok and coreGui then return coreGui end

    return Players.LocalPlayer:WaitForChild("PlayerGui")
end

local function Tween(instance, props, t)
    TweenService:Create(instance, TweenInfo.new(t or 0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), props):Play()
end

local function New(class, props)
    local inst = Instance.new(class)
    for k, v in pairs(props) do
        inst[k] = v
    end
    return inst
end

local function Corner(parent, px)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, px or 6)
    c.Parent = parent
    return c
end

local function Stroke(parent, color, thickness)
    local s = Instance.new("UIStroke")
    s.Color     = color or Color3.fromRGB(40, 40, 40)
    s.Thickness = thickness or 1
    s.Parent    = parent
    return s
end

local function ListLayout(parent, padding, dir)
    local l = Instance.new("UIListLayout")
    l.Padding       = UDim.new(0, padding or 0)
    l.FillDirection = dir or Enum.FillDirection.Vertical
    l.SortOrder     = Enum.SortOrder.LayoutOrder
    l.Parent        = parent
    return l
end

local function Padding(parent, top, bottom, left, right)
    local p = Instance.new("UIPadding")
    p.PaddingTop    = UDim.new(0, top    or 0)
    p.PaddingBottom = UDim.new(0, bottom or 0)
    p.PaddingLeft   = UDim.new(0, left   or 0)
    p.PaddingRight  = UDim.new(0, right  or 0)
    p.Parent        = parent
    return p
end

local function IsTouch(input)
    return input.UserInputType == Enum.UserInputType.Touch
end

local function IsMouse1(input)
    return input.UserInputType == Enum.UserInputType.MouseButton1
end

local function IsMouseMove(input)
    return input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch
end

local function MakeDraggable(handle, target)
    local dragging  = false
    local dragStart = Vector2.zero
    local startPos  = UDim2.new()

    handle.InputBegan:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            dragging  = true
            dragStart = Vector2.new(input.Position.X, input.Position.Y)
            startPos  = target.Position
            input:GetPropertyChangedSignal("UserInputState"):Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and IsMouseMove(input) then
            local delta = Vector2.new(input.Position.X, input.Position.Y) - dragStart
            target.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            dragging = false
        end
    end)
end

local function ConnectActivated(btn, fn)
    btn.MouseButton1Click:Connect(fn)
    btn.TouchTap:Connect(fn)
end

local ElementBuilders = {}

local function BaseElement(parent, height)
    local frame = New("Frame", {
        BackgroundColor3 = Theme.ElementBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, height or S.ElemH),
        Parent           = parent,
    })
    Corner(frame, 4)
    Stroke(frame, Theme.ElementStroke, 1)

    frame.MouseEnter:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ElementHover }) end)
    frame.MouseLeave:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ElementBg }) end)

    return frame
end

local function AccentBar(parent)
    New("Frame", {
        BackgroundColor3 = Theme.Accent,
        BorderSizePixel  = 0,
        Size             = UDim2.new(0, 3, 1, -10),
        Position         = UDim2.new(0, 0, 0, 5),
        Parent           = parent,
    })
end

function ElementBuilders.Button(parent, opts)
    local label    = opts.Label    or "Button"
    local callback = opts.Callback or function() end

    local frame = BaseElement(parent)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -20, 1, 0),
        Position               = UDim2.new(0, 14, 0, 0),
        Parent                 = frame,
    })

    local btn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = frame,
    })

    ConnectActivated(btn, function()
        Tween(frame, { BackgroundColor3 = Theme.AccentDark }, 0.08)
        task.delay(0.14, function()
            Tween(frame, { BackgroundColor3 = Theme.ElementBg }, 0.12)
        end)
        pcall(callback)
    end)
end

function ElementBuilders.Toggle(parent, opts)
    local label    = opts.Label    or "Toggle"
    local state    = opts.Default  or false
    local callback = opts.Callback or function() end

    local frame = BaseElement(parent)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -50, 1, 0),
        Position               = UDim2.new(0, 14, 0, 0),
        Parent                 = frame,
    })

    local BOX = 18
    local box = New("Frame", {
        BackgroundColor3 = state and Theme.CheckOn or Theme.CheckOff,
        BorderSizePixel  = 0,
        Size             = UDim2.new(0, BOX, 0, BOX),
        Position         = UDim2.new(1, -(BOX + 10), 0.5, -(BOX / 2)),
        Parent           = frame,
    })
    Corner(box, 3)
    Stroke(box, Theme.CheckStroke, 1)

    local mark = New("TextLabel", {
        Text                   = "✓",
        TextColor3             = Theme.CheckMark,
        Font                   = Theme.Font,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Center,
        TextYAlignment         = Enum.TextYAlignment.Center,
        BackgroundTransparency = 1,
        TextTransparency       = state and 0 or 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = box,
    })

    local btn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = frame,
    })

    ConnectActivated(btn, function()
        state = not state
        Tween(box,  { BackgroundColor3 = state and Theme.CheckOn or Theme.CheckOff }, 0.13)
        Tween(mark, { TextTransparency = state and 0 or 1 }, 0.10)
        pcall(callback, state)
    end)

    local api = {}
    function api:SetState(v)
        state = v
        box.BackgroundColor3 = v and Theme.CheckOn or Theme.CheckOff
        mark.TextTransparency = v and 0 or 1
    end
    function api:GetState() return state end
    return api
end

function ElementBuilders.Slider(parent, opts)
    local label    = opts.Label    or "Slider"
    local minVal   = opts.Min      or 0
    local maxVal   = opts.Max      or 100
    local default  = math.clamp(opts.Default or minVal, minVal, maxVal)
    local suffix   = opts.Suffix   or ""
    local callback = opts.Callback or function() end

    local frame = BaseElement(parent, S.ElemH + 14)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -65, 0, 20),
        Position               = UDim2.new(0, 14, 0, 5),
        Parent                 = frame,
    })

    local valLabel = New("TextLabel", {
        Text                   = tostring(default) .. suffix,
        TextColor3             = Theme.Accent,
        Font                   = Theme.Font,
        TextSize               = 12,
        TextXAlignment         = Enum.TextXAlignment.Right,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0, 55, 0, 20),
        Position               = UDim2.new(1, -62, 0, 5),
        Parent                 = frame,
    })

    local PX = 14
    local track = New("Frame", {
        BackgroundColor3 = Theme.SliderTrack,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, -(PX * 2), 0, 5),
        Position         = UDim2.new(0, PX, 0, 29),
        Parent           = frame,
    })
    Corner(track, 3)

    local initPct = (default - minVal) / (maxVal - minVal)
    local fill = New("Frame", {
        BackgroundColor3 = Theme.SliderFill,
        BorderSizePixel  = 0,
        Size             = UDim2.new(initPct, 0, 1, 0),
        Parent           = track,
    })
    Corner(fill, 3)

    local KS = 13
    local knob = New("Frame", {
        BackgroundColor3 = Theme.SliderKnob,
        BorderSizePixel  = 0,
        Size             = UDim2.new(0, KS, 0, KS),
        Position         = UDim2.new(initPct, -(KS / 2), 0.5, -(KS / 2)),
        Parent           = track,
    })
    Corner(knob, KS)

    local current = default
    local sliding = false

    local function Update(absX)
        local ap = track.AbsolutePosition
        local as = track.AbsoluteSize
        local pct = math.clamp((absX - ap.X) / as.X, 0, 1)
        local val = math.round(minVal + (maxVal - minVal) * pct)
        fill.Size     = UDim2.new(pct, 0, 1, 0)
        knob.Position = UDim2.new(pct, -(KS / 2), 0.5, -(KS / 2))
        valLabel.Text = tostring(val) .. suffix
        if val ~= current then
            current = val
            pcall(callback, val)
        end
    end

    track.InputBegan:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            sliding = true
            Update(input.Position.X)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if sliding and IsMouseMove(input) then
            Update(input.Position.X)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            sliding = false
        end
    end)

    local api = {}
    function api:SetValue(v)
        v = math.clamp(math.round(v), minVal, maxVal)
        local pct = (v - minVal) / (maxVal - minVal)
        fill.Size     = UDim2.new(pct, 0, 1, 0)
        knob.Position = UDim2.new(pct, -(KS / 2), 0.5, -(KS / 2))
        valLabel.Text = tostring(v) .. suffix
        current = v
    end
    function api:GetValue() return current end
    return api
end

function ElementBuilders.Dropdown(parent, opts)

    local items    = opts.Items    or {}
    local selected = opts.Default  or (items[1] or "Selecione...")
    local callback = opts.Callback or function() end

    local isOpen = false

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0.5, 0, 1, 0),
        Position               = UDim2.new(0, 14, 0, 0),
        Parent                 = frame,
    })

    local selLabel = New("TextLabel", {
        Text                   = selected,
        TextColor3             = Theme.Accent,
        Font                   = Theme.FontSemi,
        TextSize               = 12,
        TextXAlignment         = Enum.TextXAlignment.Right,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0.42, 0, 1, 0),
        Position               = UDim2.new(0.5, 0, 0, 0),
        Parent                 = frame,
    })

    local arrow = New("TextLabel", {
        Text                   = "▼",
        TextColor3             = Theme.SubText,
        Font                   = Theme.FontReg,
        TextSize               = 10,
        TextXAlignment         = Enum.TextXAlignment.Center,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0, 18, 1, 0),
        Position               = UDim2.new(1, -20, 0, 0),
        Parent                 = frame,
    })

    local ITEM_H    = 30
    local MAX_ITEMS = 5
    local listH     = math.min(#items, MAX_ITEMS) * ITEM_H

    local listFrame = New("Frame", {
        BackgroundColor3 = Theme.DropBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, listH),
        Position         = UDim2.new(0, 0, 1, 3),
        Visible          = false,
        ZIndex           = 20,
        ClipsDescendants = true,
        Parent           = frame,
    })
    Corner(listFrame, 4)
    Stroke(listFrame, Theme.ElementStroke, 1)

    local scroll = New("ScrollingFrame", {
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        Size                   = UDim2.new(1, 0, 1, 0),
        CanvasSize             = UDim2.new(0, 0, 0, #items * ITEM_H),
        ScrollBarThickness     = 3,
        ScrollBarImageColor3   = Theme.ScrollBar,
        ZIndex                 = 20,
        Parent                 = listFrame,
    })
    ListLayout(scroll, 0)

    local function CloseList()
        isOpen = false
        listFrame.Visible = false
        arrow.Text = "▼"
    end

    for _, itemText in ipairs(items) do
        local item = New("TextButton", {
            Text                   = itemText,
            TextColor3             = Theme.ElementText,
            Font                   = Theme.FontReg,
            TextSize               = 12,
            TextXAlignment         = Enum.TextXAlignment.Left,
            BackgroundColor3       = Theme.DropItem,
            BorderSizePixel        = 0,
            Size                   = UDim2.new(1, 0, 0, ITEM_H),
            ZIndex                 = 20,
            Parent                 = scroll,
        })
        Padding(item, 0, 0, 10, 0)

        item.MouseEnter:Connect(function() Tween(item, { BackgroundColor3 = Theme.DropHover }) end)
        item.MouseLeave:Connect(function() Tween(item, { BackgroundColor3 = Theme.DropItem }) end)

        ConnectActivated(item, function()
            selected = itemText
            selLabel.Text = itemText
            CloseList()
            pcall(callback, itemText)
        end)
    end

    local toggleBtn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        ZIndex                 = 10,
        Parent                 = frame,
    })

    ConnectActivated(toggleBtn, function()
        isOpen = not isOpen
        listFrame.Visible = isOpen
        arrow.Text = isOpen and "▲" or "▼"
    end)

    local api = {}
    function api:SetSelected(v) selected = v; selLabel.Text = v end
    function api:GetSelected() return selected end
    return api
end

function CrimsonUI:CreateWindow(opts)

function Window:AddTab(name)
    local btn = New("TextButton", {
        Text             = name,
        TextColor3       = Theme.TabText,
        Font             = Theme.FontSemi,
        TextSize         = 12,
        BackgroundColor3 = Theme.TabInactive,
        BorderSizePixel  = 0,
        AutoButtonColor  = false,
        Size             = UDim2.new(0, S.TabW, 1, 0),
        Parent           = self._TabBar,
    })

    local indicator = New("Frame", {
        BackgroundColor3 = Theme.TabActive,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, 2),
        Position         = UDim2.new(0, 0, 1, -2),
        Visible          = false,
        ZIndex           = 3,
        Parent           = btn,
    })

    local scroll = New("ScrollingFrame", {
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        Size                   = UDim2.new(1, 0, 1, 0),
        CanvasSize             = UDim2.new(0, 0, 0, 0),
        ScrollBarThickness     = 3,
        ScrollBarImageColor3   = Theme.ScrollBar,
        ScrollingDirection     = Enum.ScrollingDirection.Y,
        Visible                = false,
        Parent                 = self._Content,
    })

    local layout = ListLayout(scroll, S.ElemPad)
    Padding(scroll, S.SidePad, S.SidePad, S.SidePad, S.SidePad)

    layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        scroll.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + S.SidePad * 2)
    end)

    local tabObj = setmetatable({}, Tab)
    tabObj:_Init(scroll)

    local index = #self._Tabs + 1
    self._Tabs[index] = { Tab = tabObj, Scroll = scroll, Btn = btn, Indicator = indicator }

    local function SelectTab()
        for i, data in ipairs(self._Tabs) do
            local active = (i == index)
            data.Scroll.Visible      = active
            data.Indicator.Visible   = active
            data.Btn.TextColor3      = active and Theme.TabTextActive or Theme.TabText
            data.Btn.BackgroundColor3 = active and Color3.fromRGB(25,25,25) or Theme.TabInactive
        end
    end

    ConnectActivated(btn, SelectTab)

    if index == 1 then SelectTab() end

    return tabObj
end

function Window:Destroy()
    if self._Gui then self._Gui:Destroy() end
end
